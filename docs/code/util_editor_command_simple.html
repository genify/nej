    var _  = NEJ.P,
        _t = _('nej.ut'),
        _p = _('nej.ut.cmd'),
        _proSimpleCommand;
    if (!!_p._$$SimpleCommand) return;
    /**
     * 简易命令封装
     * @class   {nej.ut.cmd._$$SimpleCommand} 简易命令封装
     * @extends {nej.ut._$$EditorCommand}
     * @param   {Object} 可选配置参数，已处理参数列表如下

    _proSimpleCommand._$execute = function(){
