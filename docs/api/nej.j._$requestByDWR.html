var j = nej.j;
j._$requestByDWR('LogBean.log',
   {path:'http://123.163.com/webmail/dwr/call/plaincall/',
    script:true,param:{},onload:function(data){
        // 正常回调方法
    },onerror:function(error){

    },onbeforerequest:function(data){
       // 请求发送前对请求数据进行处理
    }

• 输入

类型：String
描述：请求地址,格式为class.method
类型：Object
描述：可选配置参数,已处理参数列表如下：
名称类型描述
pathString请求路径,默认为/dwr/call/plaincall/
queryString请求地址附带的查询参数,格式a=aaa&b=bbb
paramArray参数列表,不传或空数组均作为无参数处理
proxyString代理标识，使用该代理时忽略script/sync/method/timeout属性
scriptBoolean使用脚本方式载入
syncBoolean是否同步请求,使用脚本载入方式忽略此属性
methodString请求方式,GET/POST,使用脚本载入方式忽略此属性
timeoutNumber请求超时时间
headersObject头信息,批处理请求合并所有头信息,同名的头信息后面请求覆盖前面请求
Stringsession
