var url = "http://123.163.com/webmail/dwr/call/plaincall/DownloadBean.getDownloadUrlByBrandAndModel.dwr";
var opt = {
     param:{brand:'nokia',model:'9'}
    ,data:'123'
    ,method:'post'
    ,onload:function(_data){
        // 请求正常回调
    }
    ,onerror:function(_error){
        // 异常回调
    }
}
j._$requestByREST(url,opt);

• 输入

类型：String
描述：请求地址
类型：Object
描述：可选配置参数，已处理参数列表如下
名称类型描述
syncBoolean是否同步请求
dataVariable要发送的数据
paramObject请求参数,包括模板地址里使用的参数
methodString请求方式,GET/POST/PUT/DELETE
timeoutNumber超时时间,0 禁止超时监测
headersObject头信息
