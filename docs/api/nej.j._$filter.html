var _j = NEJ.P('nej.j');
 _j._$filter(function(_event){
        // 过滤掉404的异常，如果type是onload不做处理
    if(_event.type == 'onerror'){
        if(_event.result.data == 404)
            _event.stopped = false;
    }
});
_j._$request('xxxx',{
        type:'json',
        method:'POST',
        data:{name:'cheng-lin'},
        timeout:3000,
        onload:function(_data){
        },
        onerror:function(_error){
        }}
    );
