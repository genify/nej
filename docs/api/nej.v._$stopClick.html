var _v = NEJ.P('nej.v');
 
// 添加非点击事件
_v._$addEvent(
  'link-node-id','mousedown',
  function(_event){
      // 阻止打开链接
      _v._$stopClick(_event);
      // TODO something
  });
 
// 添加点击事件
_v._$addEvent(
  'link-node-id','click',
  function(_event){
      // 此时等价于调用 _v._$stopDefault(_event)
      _v._$stopClick(_event);
      // TODO something
  });
